//自己实现字符串拷贝函数


char *StrCpy(char* src)

{
	int len,i;

	char *result;//1、计算src的长度 len	
                len=strlen(src);

                result=(char*)malloc((len+1)*sizeof(char));//2、用malloc申请长度为 len+1的内存空间

    
                for(i=0;i<len;i++)
	
               {
		
                        result[i]=src[i];
	
	}//3、将src字符串逐个字符复制到新申请的内存空间
            

    result[i]='\0';//4、新申请内存空间的最后一位放入'\0'

    
                return result;//5、返回新申请的内存空间指针

}
